89  	public Session(String clientId, String ip, Integer port, int keepAliveSeconds, boolean cleanSession, Message will) {
90  		this.clientId = clientId;
91  		this.ip = ip;
92  		this.port = port;
93  		this.isConnected = true;
94  		this.createTime = new Date();
95  		this.currentMessageId = 0;
96  		this.keepAliveSeconds = keepAliveSeconds;
97  		this.lastIncomingTime = new Date();
98  		this.cleanSession = cleanSession;
99  		this.will = will; // [MQTT-3.1.2-9]
100 		this.topicSubscriptions = Hazelcast.INSTANCE.getMap(topicSubscriptionsName());
101 		this.topicSubscriptions.addInterceptor(new TopicSubscriptionInterceptor(clientId));
102 
103 		this.messageSender = new MessageSender(this);
104 	}

169 	public TopicSubscription matches(String topicName) {
170 		return topicSubscriptions.values().stream().filter(t -> TopicMatcher.match(t.topicFilter(), topicName))
171 				.max((p1, p2) -> p1.qos().compareTo(p2.qos())).orElse(null); // [MQTT-3.3.5-1]
172 	}

186 	public int nextMessageId() {
187 		currentMessageId = currentMessageId + 1;
188 
189 		if (currentMessageId > Message.MAX_MESSAGE_ID_NUM) {
190 			currentMessageId = Message.MIN_MESSAGE_ID_NUM;
191 		}
192 
193 		Session.NEXUS.persist(this);
194 
195 		return currentMessageId;
196 	}

198 	public void dispose(boolean sendWill) {
199 		setConnected(false);
200 
201 		if (sendWill && will != null) { // [MQTT-3.1.2-12]
202 			Topic topic = Topic.NEXUS.prepare(will);
203 			topic.publish(will);
204 
205 			will(null); // [MQTT-3.1.2-10]
206 		}
207 
208 		ChannelId channelId = null;
209 		ChannelHandlerContext ctx = NEXUS.channelHandlerContext(clientId);
210 		if (ctx != null) {
211 			ctx.channel().disconnect().addListener(ChannelFutureListener.CLOSE).addListener(fs -> Plugins.INSTANCE
212 					.get(DisconnectEventListener.class).disconnected(new AbnormalDisconnectEventArgs()));
213 
214 			channelId = ctx.channel().id();
215 		}
216 
217 		logger.debug("Session disposed [clientId={}/channelId={}]", clientId, ctx == null ? "null" : channelId);
218 
219 		if (cleanSession) {
220 			this.topicSubscriptions.values().stream().forEach(ts -> {
221 				Topic.NEXUS.matches(ts.topicFilter()).forEach(t -> t.subscribers().remove(clientId));
222 			});
223 
224 			this.topicSubscriptions.destroy();
225 		}
226 
227 		NEXUS.remove(this);
228 
229 		Plugins.INSTANCE.get(DisconnectEventListener.class).disconnected(new DisconnectEventArgs() {
230 			@Override
231 			public String clientId() {
232 				return clientId;
233 			}
234 
235 			@Override
236 			public Boolean cleanSession() {
237 				return cleanSession;
238 			}
239 
240 			@Override
241 			public Boolean byDisconnectMessage() {
242 				return !sendWill;
243 			}
244 		});
245 	}

259 	@Override
260 	public void writePortable(PortableWriter writer) throws IOException {
261 		List<String> nullChecker = Lists.newArrayList();
262 
263 		if (clientId != null) {
264 			writer.writeUTF("clientId", clientId);
265 			nullChecker.add("clientId");
266 		}
267 
268 		if (ip != null) {
269 			writer.writeUTF("ip", ip);
270 			nullChecker.add("ip");
271 		}
272 
273 		if (port != null) {
274 			writer.writeInt("port", port);
275 			nullChecker.add("port");
276 		}
277 
278 		if (isConnected != null) {
279 			writer.writeBoolean("isConnected", isConnected);
280 			nullChecker.add("isConnected");
281 		}
282 
283 		if (currentMessageId != null) {
284 			writer.writeInt("currentMessageId", currentMessageId);
285 			nullChecker.add("currentMessageId");
286 		}
287 
288 		if (will != null) {
289 			writer.writePortable("will", will);
290 		}
291 		else {
292 			writer.writeNullPortable("will", SerializableFactory.ID, Message.ID);
293 		}
294 
295 		if (cleanSession != null) {
296 			writer.writeBoolean("cleanSession", cleanSession);
297 			nullChecker.add("cleanSession");
298 		}
299 
300 		if (keepAliveSeconds != null) {
301 			writer.writeInt("keepAliveSeconds", keepAliveSeconds);
302 			nullChecker.add("keepAliveSeconds");
303 		}
304 
305 		if (createTime != null) {
306 			writer.writeLong("createTime", createTime.getTime());
307 			nullChecker.add("createTime");
308 		}
309 
310 		if (lastIncomingTime != null) {
311 			writer.writeLong("lastIncomingTime", lastIncomingTime.getTime());
312 			nullChecker.add("lastIncomingTime");
313 		}
314 
315 		writer.writeUTFArray("nullChecker", nullChecker.toArray(new String[0]));
316 	}

318 	@Override
319 	public void readPortable(PortableReader reader) throws IOException {
320 		List<String> nullChecker = Lists.newArrayList(reader.readUTFArray("nullChecker"));
321 
322 		if (nullChecker.contains("clientId")) clientId = reader.readUTF("clientId");
323 		if (nullChecker.contains("ip")) ip = reader.readUTF("ip");
324 		if (nullChecker.contains("port")) port = reader.readInt("port");
325 		if (nullChecker.contains("isConnected")) isConnected = reader.readBoolean("isConnected");
326 		if (nullChecker.contains("currentMessageId")) currentMessageId = reader.readInt("currentMessageId");
327 		will = reader.readPortable("will");
328 		if (nullChecker.contains("cleanSession")) cleanSession = reader.readBoolean("cleanSession");
329 		if (nullChecker.contains("keepAliveSeconds")) keepAliveSeconds = reader.readInt("keepAliveSeconds");
330 		if (nullChecker.contains("createTime")) createTime = new Date(reader.readLong("createTime"));
331 		if (nullChecker.contains("lastIncomingTime")) lastIncomingTime = new Date(reader.readLong("lastIncomingTime"));
332 
333 		topicSubscriptions = Hazelcast.INSTANCE.getMap(topicSubscriptionsName());
334 
335 		messageSender = new MessageSender(this);
336 	}
